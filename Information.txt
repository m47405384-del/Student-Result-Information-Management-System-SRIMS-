    function numFix(v){var n=Number(v); if(isNaN(n)||n<0) n=0; return n;}
    function gradeCalc(p){var g='F'; var r='Fail'; if(p>=90){g='A+'; r='Excellent';} else if(p>=80){g='A'; r='Very Good';} else if(p>=70){g='B'; r='Good';} else if(p>=60){g='C'; r='Average';} else if(p>=50){g='D'; r='Pass';} else{g='F'; r='Fail';} return [g,r];}

    function lookupByAdmission(){
      var adm=document.getElementById('admission').value||'';
      if(adm==='')return;
      for(var i=0;i<studentsData.length;i++){var s=studentsData[i]; if(String(s.admission)===String(adm)){fillStudentFields(s); return;}}
      var lsStudents=JSON.parse(localStorage.getItem('students')||'[]');
      for(var j=0;j<lsStudents.length;j++){var ss=lsStudents[j]; if(String(ss.admission)===String(adm)){fillStudentFields(ss); return;}}
    }

    function lookupByRoll(){
      var roll=document.getElementById('sroll').value||'';
      if(roll==='')return;
      for(var i=0;i<studentsData.length;i++){var s=studentsData[i]; if(String(s.roll)===String(roll)){fillStudentFields(s); return;}}
      var lsStudents=JSON.parse(localStorage.getItem('students')||'[]');
      for(var j=0;j<lsStudents.length;j++){var ss=lsStudents[j]; if(String(ss.roll)===String(roll)){fillStudentFields(ss); return;}}
    }

      students.push(obj); localStorage.setItem('students',JSON.stringify(students)); alert('Result saved for admission '+obj.admission);
    }
